Explicando un código de C en HTML

Código de C:


#include <stdio.h>

int main() {
    int edad;
    printf("¿Cuántos años tienes? ");
    scanf("%d", &edad);

    if (edad >= 18) {
        printf("Eres mayor de edad.\n");
    } else {
        printf("Eres menor de edad.\n");
    }

    return 0;
}
